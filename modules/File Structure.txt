📁 modules/
│

├── acr/               # Azure Container Registry
│

├── keyvault/          # Store secrets, certs, pull secrets
├── managed-identity/  # User-assigned identity for AKS & pipelines
├── storage/           # Blob storage for Terraform state or backups
├── rbac/              # Role assignments for Azure AD integration
└── bastion/           # Bastion host (optional for SSH access without public IPs)
